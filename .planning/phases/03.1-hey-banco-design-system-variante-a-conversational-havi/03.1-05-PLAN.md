    ```tsx
    "use client";

    import { useId } from "react";

    type Props = {
      data: number[];
      width?: number;
      height?: number;
      color?: string;
      fillOpacity?: number;
    };

    /**
     * Inline SVG line + area sparkline.
     *
     * Bundle reference: VariantAv2.jsx:227-235.
     * Per-instance gradient id via React useId() — multiple sparklines on
     * one page won't collide on `id="spkA"`.
     */
    export function Sparkline({
      data,
      width = 100,
      height = 40,
      color = "var(--color-hey-accent-magenta)",
      fillOpacity = 0.6,
    }: Props) {
      const gradId = `spk-${useId().replace(/:/g, "")}`;
      if (data.length < 2) return null;
      const max = Math.max(...data);
      const baseline = height - 4; // bottom padding

      const points = data
        .map((v, i) => {
          const x = (i / (data.length - 1)) * width;
          const y = baseline - (v / max) * (height - 8);
          return `${x},${y}`;
        })
        .join(" ");

      return (
        <svg width={width} height={height} viewBox={`0 0 ${width} ${height}`} className="flex-none">
          <defs>
            <linearGradient id={gradId} x1="0" y1="0" x2="0" y2="1">
              <stop offset="0" stopColor={color} stopOpacity={fillOpacity} />
              <stop offset="1" stopColor={color} stopOpacity="0" />
            </linearGradient>
          </defs>
          <polyline
            points={points}
            fill="none"
            stroke={color}
            strokeWidth="1.8"
            strokeLinecap="round"
          />
          <polygon points={`0,${height} ${points} ${width},${height}`} fill={`url(#${gradId})`} />
        </svg>
      );
    }
    ```

    Notes: `useId()` requires `"use client"`. The `:` characters in `useId()` output are stripped to keep the URL fragment valid (`url(#spk-r1)` not `url(#spk-:r1:)`).

    import { motion } from "motion/react";
    import { Icon, type IconName } from "./Icon";
    import { formatMXN } from "@/app/_data/format";

    type Props = {

    /** Locale-aware date label per UI-SPEC §"Date formatting" lines 240-247. */
    function formatTxDate(iso: string): string {
      const date = new Date(iso);
      const now = new Date();
      const diffMs = now.getTime() - date.getTime();
      const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
      const sameDay =
        date.getDate() === now.getDate() &&
        date.getMonth() === now.getMonth() &&
        date.getFullYear() === now.getFullYear();
      if (sameDay) {
        const time = new Intl.DateTimeFormat("es-MX", {
          hour: "numeric",
          minute: "numeric",
        }).format(date);
        return `Hoy ${time}`;
      }
      if (diffDays === 1) return "Ayer";
      if (diffDays <= 30) {
        return new Intl.DateTimeFormat("es-MX", {
          day: "numeric",
          month: "short",
        }).format(date);
      }
      return new Intl.DateTimeFormat("es-MX", {
        day: "numeric",
        month: "short",
        year: "numeric",
      }).format(date);
    }

    /** Map iconHint enum to a category-color fallback if Transaction.color isn't set. */
    function colorForCategory(categoryId: string): string {
      switch (categoryId) {
        case "comida":          return "var(--color-hey-accent-amber)";
        case "transporte":      return "var(--color-hey-accent-cyan)";
        case "compras":         return "var(--color-hey-accent-magenta)";
        case "servicios":       return "var(--color-hey-accent-sky)";
        case "entretenimiento": return "var(--color-hey-accent-purple)";
        default:                return "var(--color-hey-accent-green)";
      }
    }

    /** Resolve an Icon name from a transaction's iconHint (closed union match). */
    function iconForHint(hint: Transaction["iconHint"]): IconName {
      switch (hint) {
        case "coffee":   return "coffee";
        case "uber":     return "uber";
        case "bag":      return "bag";
        case "card":     return "card";
        case "transfer": return "transfer";
        case "wallet":   return "wallet";
        default:         return "card";
      }
    }

    import { useEffect, useRef } from "react";
    import type { ChatMessage as ChatMessageType } from "@/app/_data/types";
    import { ChatMessage } from "./ChatMessage";
    import { TypingIndicator } from "./TypingIndicator";

    type Props = {
      messages: ChatMessageType[];
      typing: boolean;
      onNavigate: (href: string) => void;
      onAgent: () => void;
    };

    /**
     * Scrollable container of chat messages.
     * ARIA: role="log" + aria-live="polite" + aria-relevant="additions" so
     * assistive tech announces new messages without re-reading history.
     */
    export function MessagesList({ messages, typing, onNavigate, onAgent }: Props) {
      const scrollerRef = useRef<HTMLDivElement>(null);

      // Auto-scroll to bottom on every render where the message count or typing
      // state changes. Pure scrollTop assignment — cheap, no animation.
      useEffect(() => {
        const el = scrollerRef.current;
        if (el) el.scrollTop = el.scrollHeight;
      }, [messages.length, typing]);

      return (
        <div
          ref={scrollerRef}
          role="log"
          aria-live="polite"
          aria-relevant="additions"
          aria-label="Conversación con HAVI"
          className="flex-1 overflow-y-auto px-4 py-3 flex flex-col gap-3.5"
        >
          {messages.map((m) => (
            <ChatMessage key={m.id} msg={m} onNavigate={onNavigate} onAgent={onAgent} />
          ))}
          {typing && <TypingIndicator />}
        </div>
      );
    }

    Notes:
    - `aria-relevant="additions"` (UI-SPEC line 587) tells screen readers to announce only new entries, not re-read history on every state change. Pairs with `aria-live="polite"` so announcement isn't urgent (interrupting).
