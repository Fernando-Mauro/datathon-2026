    ```tsx
    "use client";

    import { use } from "react";
    import { PageTransition } from "@/app/_components/PageTransition";
    import { AppHeader } from "@/app/_components/AppHeader";
    import { BarChart } from "@/app/_components/BarChart";
    import { mockCategories, mockSpentThisMonth } from "@/app/_data/mock";

    type Params = { categoria: string };

    export default function GraficaPage({ params }: { params: Promise<Params> }) {
      const { categoria } = use(params); // Next 16 K-8 — params is a Promise

      // For "general" or any other categoria value, show ALL categories.
      // Future enhancement: filter by exact categoria slug ("comida" → only that).
      // v1 design: BarChart always renders the full breakdown — the URL just
      // disambiguates entry points (e.g. /app/grafica/comida from a category chip).
      const segments = mockCategories
        .filter((c) => c.spent > 0)
        .map((c) => {
          const total = mockCategories.reduce((sum, x) => sum + x.spent, 0);
          const pct = total > 0 ? Math.round((c.spent / total) * 100) : 0;
          return {
            id: c.id,
            label: c.name,
            amount: c.spent,
            pct,
            color: c.color,
          };
        });
      const totalValue = segments.reduce((sum, s) => sum + s.amount, 0);

      return (
        <PageTransition from="right">
          <main className="hey-app-frame flex min-h-screen flex-col">
            <AppHeader variant="back" title="Gasto por categoría" />
            <section className="flex-1 overflow-y-auto px-5 py-5">
              <BarChart segments={segments} totalLabel="ESTA SEMANA" totalValue={totalValue} />
            </section>
          </main>
        </PageTransition>
      );
    }
    ```

    Critical:
    - `params: Promise<Params>` (Next 16 K-8) — resolved via `use(params)` from React. NOT `await` (that's for server components only; this file is `"use client"`).

    - `mockSpentThisMonth` is intentionally NOT the `totalValue` — `totalValue` is computed from the segments (sum of `c.spent`) so the bar visually adds up to 100%. If they diverge, the design's eyebrow "ESTA SEMANA $3,280.50" looks wrong because the bar segments add to less. Compute from segments to keep visuals consistent.
