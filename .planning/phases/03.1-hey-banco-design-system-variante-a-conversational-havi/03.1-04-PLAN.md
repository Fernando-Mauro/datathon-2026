    ```tsx
    export function PageTransition({
      children,

          transition={{ duration: 0.32, ease: [0.32, 0.72, 0, 1] }}
    ```

    ```ts
    "use client";

    /**
     * Phase 03.1 D-47 — chat persistence + sign-out clear.
     *
     * Functions:
     *   - loadChat(): ChatMessage[] | null — read on mount; SSR-safe (returns null server-side)
     *   - saveChat(messages): debounced write target (caller debounces ~250ms)
     *   - useSignOutClear(): subscribes to Hub.listen("auth", "signedOut") → clears KEY
     *
     * Storage key:  "havi-chat-v1"  (version part of key — bump v2 → invalidate)
     * Shape:        { version: 1, messages: ChatMessage[], updatedAt: ISO }
     * Version:      mismatch on read → wipe key + return null (cold start reseed)
     *
     * Pitfall 8: Hub.listen returns a cleanup fn directly — pass it as the
     *            useEffect cleanup return value. NEVER return () => {}.
     */
    import { useEffect } from "react";
    import { Hub } from "aws-amplify/utils";
    import type { ChatMessage } from "@/app/_data/types";

    const KEY = "havi-chat-v1";
    const VERSION = 1;

    export interface PersistedChat {
      version: number;
      messages: ChatMessage[];
      updatedAt: string;
    }

    export function loadChat(): ChatMessage[] | null {
      if (typeof window === "undefined") return null;
      try {
        const raw = localStorage.getItem(KEY);
        if (!raw) return null;
        const parsed = JSON.parse(raw) as PersistedChat;
        if (parsed.version !== VERSION) {
          localStorage.removeItem(KEY);
          return null;
        }
        return parsed.messages;
      } catch {
        // corrupt JSON or quota error — cold start
        return null;
      }
    }

    export function saveChat(messages: ChatMessage[]): void {
      if (typeof window === "undefined") return;
      try {
        const payload: PersistedChat = {
          version: VERSION,
          messages,
          updatedAt: new Date().toISOString(),
        };
        localStorage.setItem(KEY, JSON.stringify(payload));
      } catch {
        // quota exceeded — silently drop (best-effort persistence in v1)
      }
    }

    export function useSignOutClear(): void {
      useEffect(() => {
        const stop = Hub.listen("auth", ({ payload }) => {
          if (payload.event === "signedOut") {
            localStorage.removeItem(KEY);
          }
        });
        return stop; // Hub.listen returns the cleanup fn directly (Pitfall 8)
      }, []);
    }
    ```
