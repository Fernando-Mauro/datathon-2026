      mockBalance,
      mockSpentThisMonth,
      mockSpentPct,
      mockSparkline,

    /** localStorage key for cached firstName (Phase 03.1 D-50). */

    /** Suggestion chips canned-prompts (UI-SPEC §Copywriting Contract line 705). */

     *   attrs.name → first whitespace-token → email local-part → "amigo"

    function deriveFirstName(name?: string, email?: string): string {
      const fromName = name?.trim().split(/\s+/)[0];
      if (fromName) return fromName;
      if (email) return email.split("@")[0];
      return "amigo";
    }

    /** Seed messages for cold start (no localStorage). */
    function seedIntroMessages(firstName: string): ChatMessage[] {
      const now = new Date().toISOString();
      return [
        {
          id: "intro-1",
          from: "havi",
          kind: "intro",
          text: `hey, ${firstName}.`,
          createdAt: now,

          id: "snapshot-1",

            spent: mockSpentThisMonth,
            spentPct: mockSpentPct,
            topCategory: "Comida",
            sparkline: mockSparkline,
          },
          createdAt: now,
        },
      ];
    }

    /** Generate a probably-unique message id without crypto.randomUUID assumptions. */
    function generateId(prefix: string): string {
      return `${prefix}-${Date.now().toString(36)}-${Math.random().toString(36).slice(2, 8)}`;
    }

    export default function ChatHome() {
      const router = useRouter();
      const { user } = useAuthenticator(({ user }) => [user]);
      const email = user?.signInDetails?.loginId;

      // ---- Greeting state ----------------------------------------------------
      const [firstName, setFirstName] = useState<string>(() => {
        if (typeof window === "undefined") return "amigo";
        return localStorage.getItem(FIRST_NAME_KEY) ?? "amigo";
      });

      // Background refresh of firstName from Cognito (RESEARCH §7).
      // Mirrors Phase 3 placeholder's fetchUserAttributes idiom (cancellation flag).
      useEffect(() => {
        let cancelled = false;
        (async () => {
          try {
            const attrs = await fetchUserAttributes();

            if (!cancelled) {
              setFirstName(fn);
              if (typeof window !== "undefined") {
                localStorage.setItem(FIRST_NAME_KEY, fn);
              }
            }
          } catch (e) {
            if (process.env.NODE_ENV !== "production") {
              console.warn("[ChatHome] fetchUserAttributes failed:", e);
            }
          }
        })();
        return () => {
          cancelled = true;
        };
      }, [email]);

        const persisted = typeof window !== "undefined" ? loadChat() : null;
        if (persisted && persisted.length > 0) return persisted;

        // below will rewrite the intro text in-place once Cognito attrs land.
        const seed =

            return updated;
          }
          return prev;
        });

        return () => {
          if (debounceRef.current) clearTimeout(debounceRef.current);

      // ---- Send loop ---------------------------------------------------------
      function send(text: string) {
        const trimmed = text.trim();
        if (!trimmed) return;
        setInput("");

          id: generateId("user"),
          from: "user",
          kind: "user-text",
          text: trimmed,
          createdAt: new Date().toISOString(),

        const matched = haviPatterns.find((p) => {
          if (typeof p.match === "function") return p.match(trimmed);
          return p.match.test(trimmed);
        });
        if (!matched) {
          // Defensive — should never happen because catch-all /.*/u is the LAST entry.
          setTyping(false);
          return;

        const response = matched.respond(trimmed);
        const delay = response.delayMs ?? Math.floor(800 + Math.random() * 700);

        setTimeout(() => {
          setTyping(false);
          setMessages((m) => [...m, ...response.messages]);
        }, delay);

      function onNavigate(href: string) {
        router.push(href);
      }

      function onAgent() {
        router.push("/app/agente");
      }

      return (

          <AppHeader variant="havi" />

            messages={messages}
            typing={typing}
            onNavigate={onNavigate}

            value={input}
            onChange={setInput}
            onSubmit={send}
            placeholder="Pregúntale a HAVI..."
          />
        </main>
      );
    }
